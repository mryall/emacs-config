#+BEGIN_SRC emacs-lisp

  ;; be quiet
  (setq inhibit-startup-message  t)

  ;; clear the furniture
  (if (fboundp 'scroll-bar-mode) (scroll-bar-mode -1))
  (if (fboundp 'tool-bar-mode) (tool-bar-mode -1))
  (if (fboundp 'menu-bar-mode) (menu-bar-mode -1))
